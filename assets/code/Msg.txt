type Msg
    = WindowResizes Device.Size
      -- pointer events
    | PointerMsg Pointer.Msg
      -- select things
    | SelectImage Int
    | SelectTool Int
    | SelectClass Int

    | LoadImages (List { name : String, file : Value })

    | ConfigLoaded String

    | ZoomMsg ZoomMsg
